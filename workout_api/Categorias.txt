1. models.py

from sqlalchemy import Column, String, Integer
from sqlalchemy.orm import relationship
from workout_api.configs.database import Base

class CategoriaModel(Base):
    __tablename__ = 'categorias'

    pk_id = Column(Integer, primary_key=True, autoincrement=True)
    nome = Column(String(10), unique=True, nullable=False)
    
    # Relação: Uma categoria pode ter vários atletas (Ex: Categoria "Elite" tem 20 atletas)
    atletas = relationship("AtletaModel", back_populates="categoria")


2. schemas.py

from pydantic import BaseModel, Field
from typing import Annotated

class CategoriaIn(BaseModel):
    nome: Annotated[str, Field(description='Nome da categoria', example='Scale', max_length=10)]

class CategoriaOut(CategoriaIn):
    id: int # O ID que será retornado após a criação
