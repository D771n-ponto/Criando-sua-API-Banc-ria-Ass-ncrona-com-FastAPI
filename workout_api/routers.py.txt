from fastapi import APIRouter
# Importamos as rotas de cada módulo (você criará estas variáveis nos controllers)
from workout_api.atleta.controller import router as atleta_router
from workout_api.categorias.controller import router as categorias_router
from workout_api.centro_treinamento.controller import router as ct_router

api_router = APIRouter()

# Aqui "montamos" as rotas na API principal com prefixos e tags para o Swagger (documentação)
api_router.include_router(atleta_router, prefix="/atletas", tags=["atletas"])
api_router.include_router(categorias_router, prefix="/categorias", tags=["categorias"])
api_router.include_router(ct_router, prefix="/centros_treinamento", tags=["centros_treinamento"])
